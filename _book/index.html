API文档编写说明

目录结构遵循规则

├── books                             # 文档根目录
      ├── api1                        # 某一模块的API 的根目录（文件名可修改，可以有多个与api1同级的文件夹）
      │     │── README.md             # 该目录标题 （必须有，否则会自动取文件名作为目录名，文件中以 `# 标题` 的格式写标题名称即可）
      │     └── 文件夹或markdown文件   # api1模块的api 文件
      │     └── .gitignore            # 提交忽略文件
      │     └── book.json             # gitbook配置文件
      │     └── README.md             # gitbook首页（这里不作为首页显示，只用作该文档的说明文件）
      │     └── SUMMARY.md            # 文档的目录文件（不需要手动配置，会自动生成）

├── pages                          # api以外的门户页面内容
    ├── build                      # 构建相关
    ├── mock                       # 项目mock 模拟数据
    ├── public                     # 静态资源
    │   │── favicon.ico            # favicon图标
    │   └── index.html             # html模板
    ├── src                        # 源代码
    │   ├── api                    # 所有请求
    │   ├── assets                 # 主题 字体 图片等静态资源
    │   ├── components             # 全局公用组件
    │   ├── directive              # 全局指令
    │   ├── icons                  # 项目所有 svg icons
    │   ├── layout                 # 全局 layout
    │   ├── router                 # 路由
    │   ├── store                  # 全局 store管理
    │   ├── styles                 # 全局样式
    │   ├── utils                  # 全局公用方法
    │   ├── views                  # views 所有页面
    │   ├── App.vue                # 入口页面
    │   ├── main.js                # 入口文件 加载组件 初始化等
    │   └── permission.js          # 权限管理
    ├── tests                      # 测试
    ├── .env.xxx                   # 环境变量配置
    ├── .eslintrc.js               # eslint 配置项
    ├── ..eslintignore.js          # eslint 配置项忽略配置
    ├── .babelrc                   # babel-loader 配置
    ├── .travis.yml                # 自动化CI配置
    ├── vue.config.js              # vue-cli 配置
    ├── postcss.config.js          # postcss 配置
    └── package.json               # package.json

构建项目命令行

# 进入项目目录
cd books

# 配置gitbook环境
npm install gitbook-cli -g

# 初始化项目
gitbook init

# 安装依赖
gitbook install

# 启动服务
gitbook serve

# 构建静态网页
gitbook build
